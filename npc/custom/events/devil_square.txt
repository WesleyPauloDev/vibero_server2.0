prontera,147,154,7	script	Guardião do Devil Square#1	406,{
	set .@gm_level, 99; // Nível de GM necessário para iniciar/parar o evento

	mes "[Guardião do Devil Square]";
	mes "Esta é a entrada do ^FF0000Devil Square^000000.";
	mes "Se vocę falhar ou morrer, o portão será fechado e vocę deverá aguardar até o próximo evento para entrar novamente.";
	next;
		.@is_gm_menu = (getgmlevel() >= .@gm_level);
		.@menu$ = "Entrar.:Informacoes.:Cancelar.";
		if (.@is_gm_menu)
			.@menu$ = "Entrar.:Informacoes.:Menu GM.:Cancelar.";
		switch(select(.@menu$)) {

	case 1:
		if (.start == 0) {
			mes "[Guardião do Devil Square]";
			mes "O Devil Square está fechado no momento.";
			mes "Ele abrirá nos seguintes horários:";
			mes "00:00 , 12:00 , 18:00";
			close;
		}
		if (.start == 2) {
			mes "[Guardião do Devil Square]";
			mes "O Devil Square já iniciou.";
			close;
		}
		if (BaseLevel < 70) {
			mes "[Guardião do Devil Square]";
			mes "Vocę não atende aos requisitos para entrar no Devil Square.";
			mes " ";
			mes "Requisitos:";
			mes " - Nível Base 70";
			close;
		}
		// =========================
		// VERIFICA IP DUPLICADO
		// =========================
		.@my_ip$ = getcharip();
		.@unit_count = getmapunits(BL_PC, "ordeal_1-1", .@units[0]);
		for (.@i = 0; .@i < .@unit_count; .@i++) {
			.@other_ip$ = getcharip(.@units[.@i]);
			if (.@other_ip$ != "" && .@other_ip$ == .@my_ip$) {
				dispbottom "Ja existe um jogador conectado com este IP no Devil Square.";
				dispbottom "Apenas um personagem por IP e permitido.";
				close;
			}
		}

		announce strcharinfo(0)+" entrou no Devil Square.", bc_npc;
		percentheal 100,100;
		warp "ordeal_1-1",183,182;
		close;

	case 2:
		mes "[Guardião do Devil Square]";
		mes "Lá dentro, vocę enfrentará 3 ondas de monstros.";
		mes "A cada rodada, os inimigos ficam mais fortes.";
		next;
		mes "[Guardião do Devil Square]";
		mes "Se derrotar todos os inimigos, vocę poderá abrir 10 Baús do Tesouro contendo itens, equipamentos e cartas raras!";
		close;
		case 3:
			if (!.@is_gm_menu)
				break;
			mes "[Guardiao do Devil Square]";
			switch (.start) {
				case 0: mes "O Devil Square ainda nao comecou."; break;
				case 1: mes "O Devil Square esta aceitando participantes."; break;
				case 2: mes "O evento esta em andamento na rodada "+ .round +" com "+ .mob +" monstros restantes."; break;
			}
			next;
			switch(select("Iniciar evento.:Cancelar.")) {
			case 1:
				mes "[Guardiao do Devil Square]";
				if (.start)
					mes "O Devil Square ja foi iniciado.";
				else {
					mes "Iniciando Devil Square...";
					.gm_quick_start = 1;
					donpcevent strnpcinfo(0)+"::OnStartEvent";
				}
				close;
			case 2:
				break;
			}
			break;

	case 4:
		break;
	}

	mes "[Guardião do Devil Square]";
	mes "Até breve.";
	close;

OnStartEvent:
OnClock0000:
OnClock1200:
OnClock1800:

	.start = 1;
	// disablenpc "Guardião do Devil Square#2";
	disablenpc "Devil Square Reward Box";
	disablenpc "ord11-12";
	// mapannounce "ordeal_1-1","O Devil Square está fechado.", bc_map;

	// getmapxy .@map$, .@x, .@y, BL_NPC;
	// mapwarp "ordeal_1-1", .@map$, .@x, .@y;
	killmonsterall "ordeal_1-1";
		if (.gm_quick_start) {
			announce "O Devil Square esta ABERTO! O evento comecara em 1 minuto.", bc_all;
			sleep 60000;
			.gm_quick_start = 0;
		} else {
			announce "O Devil Square esta ABERTO! O evento comecara em 5 minutos.", bc_all;
			sleep 240000;
			announce "O Devil Square comecara em 1 minuto.", bc_all;
			sleep 60000;
		}

	announce "O Devil Square COMEĮOU!", bc_all;

	.start = 2;
	.mob = 62;
	if (!getmapusers("ordeal_1-1")) {
		.start = .round = .mob = 0;
		announce "Devil Square cancelado por falta de participantes.", bc_all;
		end;
	}
	
	.round = 1;	

	areamonster "ordeal_1-1",183,182,246,244,"[DS] SOLDADO ESQUELETO",1028,10,strnpcinfo(0)+"::OnDevilDead";
	areamonster "ordeal_1-1",183,182,246,244,"[DS] ARQUEIRO ORC",1189,10,strnpcinfo(0)+"::OnDevilDead";
	areamonster "ordeal_1-1",183,182,246,244,"[DS] SANDMAN",1165,10,strnpcinfo(0)+"::OnDevilDead";
	areamonster "ordeal_1-1",183,182,246,244,"[DS] ARQUEIRO ESQUELETO",1016,10,strnpcinfo(0)+"::OnDevilDead";
	areamonster "ordeal_1-1",183,182,246,244,"[DS] MINOROUS",1149,10,strnpcinfo(0)+"::OnDevilDead";
	areamonster "ordeal_1-1",183,182,246,244,"[DS] REQUIEM",1164,10,strnpcinfo(0)+"::OnDevilDead";
	areamonster "ordeal_1-1",183,182,246,244,"[DS] MOONLIGHT FLOWER",1150,1,strnpcinfo(0)+"::OnDevilDead";
	areamonster "ordeal_1-1",183,182,246,244,"[DS] MISTRESS",1059,1,strnpcinfo(0)+"::OnDevilDead";

	end;

OnDevilDead:
	.mob--;
	if (.mob > 0)
		announce "Devil Square: Restam "+ .mob +" monstros na rodada 1.", bc_map;
	else if (.mob <= 0) {
		announce "Devil Square está entrando na rodada 2...", bc_map,"0xFF0000";
		sleep 5000;
		goto OnDevil2;
	}
	end;

OnDevil2:
	.round = 2;
	.mob = 62;

	areamonster "ordeal_1-1",183,182,246,244,"[DS] ALARM",1193,10,strnpcinfo(0)+"::OnDevilDead2";
	areamonster "ordeal_1-1",183,182,246,244,"[DS] HIGH ORC",1213,10,strnpcinfo(0)+"::OnDevilDead2";
	areamonster "ordeal_1-1",183,182,246,244,"[DS] INJUSTICE",1257,10,strnpcinfo(0)+"::OnDevilDead2";
	areamonster "ordeal_1-1",183,182,246,244,"[DS] WRAITH",1192,10,strnpcinfo(0)+"::OnDevilDead2";
	areamonster "ordeal_1-1",183,182,246,244,"[DS] GÁRGULA",1253,10,strnpcinfo(0)+"::OnDevilDead2";
	areamonster "ordeal_1-1",183,182,246,244,"[DS] MIMIC",1191,10,strnpcinfo(0)+"::OnDevilDead2";
	areamonster "ordeal_1-1",183,182,246,244,"[DS] DARK LORD",1272,1,strnpcinfo(0)+"::OnDevilDead2";
	areamonster "ordeal_1-1",183,182,246,244,"[DS] ORK HERO",1087,1,strnpcinfo(0)+"::OnDevilDead2";

	end;

OnDevilDead2:
	.mob--;
	if (.mob > 0)
		announce "Devil Square: Restam "+ .mob +" monstros na rodada 2.", bc_map;
	else if (.mob <= 0) {
		announce "Devil Square está entrando na rodada final...", bc_map,"0xFF0000";
		sleep 5000;
		goto OnDevil3;
	}
	end;

OnDevil3:
	.round = 3;
	.mob = 42;

	areamonster "ordeal_1-1",183,182,246,244,"[DS] DEVIRUCHI",1109,10,strnpcinfo(0)+"::OnDevilDead3";
	areamonster "ordeal_1-1",183,182,246,244,"[DS] ANOLIAN",1206,10,strnpcinfo(0)+"::OnDevilDead3";
	areamonster "ordeal_1-1",183,182,246,244,"[DS] WANDERER",1208,10,strnpcinfo(0)+"::OnDevilDead3";
	areamonster "ordeal_1-1",183,182,246,244,"[DS] JOKER",1131,10,strnpcinfo(0)+"::OnDevilDead3";
	areamonster "ordeal_1-1",183,182,246,244,"[DS] BAPHOMET",1039,1,strnpcinfo(0)+"::OnDevilDead3";
	areamonster "ordeal_1-1",183,182,246,244,"[DS] VALKYRIE RANDGRIS",1751,1,strnpcinfo(0)+"::OnDevilDead3";
	end;

OnDevilDead3:
	.mob--;
	if (.mob > 0)
		announce "Devil Square: Restam "+ .mob +" monstros na rodada final.", bc_map;
	else if (.mob <= 0) {
		announce "[Devil Square] Todos os monstros foram derrotados!", bc_all,"0xFF0000";
		
		sleep 3000;
		goto OnDevil4;
	}
	end;

OnDevil4:

	enablenpc "Devil Square Reward Box";
	killmonsterall "ordeal_1-1";
	.start = .round = .mob = 0;
	end;

OnClock0030:
OnClock1230:
OnClock1830:
	if (.start == 2 && .mob > 0)
		announce "O Devil Square foi encerrado. Ninguém foi capaz de Dominar os Monstros", bc_all;
	.start = .round = .mob = 0;
	killmonsterall "ordeal_1-1";
	if (getmapusers("ordeal_1-1"))
		mapwarp "ordeal_1-1","prontera",155,187;
	
	end;
