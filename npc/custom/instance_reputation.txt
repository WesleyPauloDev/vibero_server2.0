//=============================================================
// NPC: Aumento de Reputaçăo
// Autor original: (vocę)
// Revisado por: ChatGPT
//=============================================================
prontera,160,180,4	script	Reputation Increaser	4_EP19_IWIN,{

	// ===== CONFIGURAÇŐES =====
	.@zeny_cost      = 1;    // Zeny necessários por investimento
	.@zeny_reputation = -6000;        // Reputaçăo ganha por investimento em Zeny
	.@item_cost      = 5;          // Quantidade de item por investimento
	.@item_reputation = 100;       // Reputaçăo ganha por investimento em item
	.@item_id        = 7179;       // ID do item (POD)

	// Lista de reputaçőes:
	// <ID da reputaçăo>, "<Nome>", <Limite máximo>
	setarray .@rep$, 
        1,"Instâncias",3000,
        2,"Votos",3000,
        3,"Cheffęnia Normal",5000,
        4,"Cheffęnia Hard",5000,
        5,"Bio 5 Normal",5000,
        6,"Bio 5 Hard",5000,
        7,"Reborn",5000;

	// ===== INÍCIO =====
	disable_items;
	cutin "ep19_iwin04.png",2;

	mes "[Aumento de Reputaçăo]";
	// set .@reputation, get_reputation_points(atoi(.@rep$[0]));
	// .@zeny_reputation = .@zeny_reputation - .@reputation;
	// mes "Reputaçăo: " + .@reputation;
	if(getarraysize(.@rep$) > 3)
		mes "Olá, posso ^FF8000aumentar sua reputaçăo^000000 em várias áreas usando ^FF8000Zeny ou POD^000000.";
	else
		mes "Olá, posso ^FF8000aumentar sua reputaçăo^000000 em "+.@rep$[1]+" usando ^FF8000Zeny ou POD^000000.";
	next;

	switch(select("Como funciona?","Investir Zeny","Investir POD","Sair")) {
	case 1:
		mes "[Aumento de Reputaçăo]";
		mes "Para cada doaçăo, sua reputaçăo sobe na área escolhida.";
		if(getarraysize(.@rep$)>3) {
			mes "Áreas disponíveis:";
			for(.@i=0; .@i<getarraysize(.@rep$); .@i+=3)
				mes "- "+.@rep$[.@i+1];
		}
		next;
		mes "Cada "+F_InsertComma(.@zeny_cost)+"z = "+.@zeny_reputation+" pontos, ou "+
		    .@item_cost+" "+mesitemlink(.@item_id)+" = "+.@item_reputation+" pontos.";
		close;
	case 2: .@use_zeny = true; break;
	case 3: .@use_item = true; break;
	default:
		cutin "",255;
		close;
	}

	// ===== Escolher área =====
	mes "[Aumento de Reputaçăo]";
	mes "Escolha a área:";
	.@menu$="";
	for(.@i=0; .@i<getarraysize(.@rep$); .@i+=3)
		.@menu$ += .@rep$[.@i+1]+":";
	.@s = (select(.@menu$)-1)*3;
	next;

	// ===== Verificar limite atual =====
	if(get_reputation_points(atoi(.@rep$[.@s])) >= atoi(.@rep$[.@s+2])) {
		mes "Sua reputaçăo em "+.@rep$[.@s+1]+" já atingiu o máximo.";
		cutin "",255;
		close;
	}

	// ===== Calcular quantidade =====
	mes "[Aumento de Reputaçăo]";
	mes "Quantas vezes deseja investir? (0 para cancelar)";
	input .@times;
	if(.@times < 1) close;

	// Ajustar se năo tiver recursos
	if(.@use_zeny) {
		if(.@times * .@zeny_cost > Zeny)
			.@times = Zeny / .@zeny_cost;
	} else {
		if(.@times * .@item_cost > countitem(.@item_id))
			.@times = countitem(.@item_id) / .@item_cost;
	}
	if(.@times < 1) {
		mes "Vocę năo possui recursos suficientes.";
		cutin "",255;
		close;
	}

	// ===== Confirmar =====
	.@gain = (.@use_zeny) ? .@zeny_reputation * .@times : .@item_reputation * .@times;
	if(get_reputation_points(atoi(.@rep$[.@s])) + .@gain > atoi(.@rep$[.@s+2])) {
		mes "Essa quantidade excederia o máximo permitido.";
		cutin "",255;
		close;
	}

	mes "Confirma investir para ganhar ^FF0000"+F_InsertComma(.@gain)+
	    "^000000 pontos em "+.@rep$[.@s+1]+"?";
	if(select("Cancelar","Confirmar") == 1) close;

	// ===== Cobrar e adicionar reputaçăo =====
	if(.@use_zeny) {
		if(Zeny < .@times * .@zeny_cost) { mes "Zeny insuficiente."; close; }
		Zeny -= .@times * .@zeny_cost;
	} else {
		if(countitem(.@item_id) < .@times * .@item_cost) { mes "Itens insuficientes."; close; }
		delitem .@item_id, .@times * .@item_cost;
	}
	add_reputation_points(atoi(.@rep$[.@s]), .@gain);

	// ===== Fim =====
	mes "Sua reputaçăo em "+.@rep$[.@s+1]+" aumentou em "+F_InsertComma(.@gain)+" pontos!";
	specialeffect2 EF_STEALCOIN;
	emotion ET_DELIGHT;
	cutin "",255;
	close;
}
