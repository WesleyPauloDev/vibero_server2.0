	mes "[Examinador]";
	
	if (Class == Job_Novice && JobLevel < 10) {
		mes "Vocę precisa atingir o nível 10 de classe para prosseguir.";		
		close;
		
	} else {
		mes "Vejo que vocę adquiriu bastante Experięncia de Batalha!";
		mes "Está pronto para seguir sua jornada.";
		next;
		
		if (select("Sim, me leve para Prontera.:Ainda năo.") == 1) {
			mes "[Examinador]";
			mes "Boa sorte em sua aventura!";
			close2;
			warp "prontera", 156, 191; // Coordenadas centrais de Prontera
			end;
		} else {
			mes "[Examinador]";
			mes "Tudo bem, volte quando estiver pronto.";
		}
		close;
	}
	OnInit:
		waitingroom "Examinador",0;
		end;
}

new_1-1,60,123,4	script	Quest do Iniciante	810,{

	// quest_poring_done = 0;
	// quest_poring_started = 1;
	// quest_poring_count = 0;
	if (quest_poring_done) {
		mes "[Quest Iniciante]";
		mes "Vocę já completou essa missăo.";
		close;
	}

	if (quest_poring_started == 0) {
		mes "[Quest Iniciante]";
		mes "Bem-vindo ŕ Área de Treinamento!";
		mes "Gostaria de me ajudar matando 20 Porings?";
		mes "";
		mes "[Recompensa]";
		mes "- 100 Poçőes Vermelhas";
		mes "- 100 Poçőes Azuis";
		mes "- 5000 de experięncia base";
		next;
		if (select("Sim:Năo") == 2) close;

		quest_poring_started = 1;
		quest_poring_count = 0;
		mes "[Quest Iniciante]";
		mes "Boa sorte! Volte quando tiver derrotado 20 Porings.";
		close;
	}

	if (quest_poring_started == 1 && quest_poring_count < 20) {
		mes "[Quest Iniciante]";
		mes "Vocę ainda năo matou 20 Porings.";
		mes "Progresso: " + quest_poring_count + " / 20";
		close;
	}

	if (quest_poring_started == 2 && quest_poring_count >= 20) {
		mes "[Quest Iniciante]";
		mes "Parabéns! Vocę completou a missăo.";
		mes "Aqui está sua recompensa:";
		mes "- 100 Poçőes Vermelhas";
		mes "- 100 Poçőes Azuis";
		mes "- 5000 de experięncia base";
		getitem 501, 100; // Poçăo Vermelha do iniciante
		getitem 505, 100; // Poçăo Azul do iniciante
		getexp 5000,0;      // Recompensa de experięncia base
		quest_poring_done = 1;
		quest_poring_started = 0;
		quest_poring_count = 0;
		close;
	}
}
