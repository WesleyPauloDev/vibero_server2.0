prontera,136,168,4	script	Receber VIP	4_M_MANAGER,{
	mes "[VIP]";
	
	// Já recebeu?
	if (#Reward_Vip) {
		mes "Vocę já recebeu o ^00FF00VIP^000000 nesta conta.";
		mes "Cada conta pode receber apenas uma vez.";
		close;
	}

	// Checar peso antes de entregar
	if (!checkweight(9616,1)) {
		mes "Vocę está com peso ou itens demais no inventário.";
		mes "Guarde alguns itens e tente novamente.";
		close;
	}

	// Entregar VIP
	getitem 9616,1;
	set #Reward_Vip, 1;

	mes "^00FF00VIP entregue com sucesso!^000000";
	mes "Aproveite seu benefício.";
	close;
}
