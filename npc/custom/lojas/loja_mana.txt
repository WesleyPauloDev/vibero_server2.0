// Slotador de Botas Temporais
-	script	Hugins Craftsman#pa0829	4_F_JOB_BLACKSMITH,{

    OnTalk:
        disable_items;
        if (MaxWeight - Weight < 1000) {
            mes "Sua bolsa está muito pesada. Reduza um pouco o peso e volte depois.";
            close;
        }
        mes "[Artesăo de Hugin]";
        mes "Traga-me ^0000FF20 Cristais Temporais e 50 Manas Coaguladas^000000, e eu tentarei colocar um slot nas suas Botas Temporais.";
        mes "Mas saiba que existe 20% de chance de ^FF0000falhar e quebrar o item^000000. Deseja tentar mesmo assim?";
        next;
        if (select("Aceitar o risco e tentar colocar o slot", "Cancelar") == 0) {
            mes "[Artesăo de Hugin]";
            mes "Tudo bem, volte quando quiser.";
            close;
        }
        if (!getequipisequiped(EQI_SHOES)) {
            mes "[Artesăo de Hugin]";
            mes "Tem certeza de que está usando o item?";
            close;
        }

        .@equip_id = getequipid(EQI_SHOES);
        switch(.@equip_id) {
        case 22000:	callsub(S_Slot,22006);	break; // Temporal_STR_Boots
        case 22001:	callsub(S_Slot,22009);	break; // Temporal_INT_Boots
        case 22002:	callsub(S_Slot,22010);	break; // Temporal_AGI_Boots
        case 22003:	callsub(S_Slot,22007);	break; // Temporal_VIT_Boots
        case 22004:	callsub(S_Slot,22008);	break; // Temporal_DEX_Boots
        case 22005:	callsub(S_Slot,22011);	break; // Temporal_LUK_Boots

        case 22107:	callsub(S_Slot,22113);	break; // Modified_STR_Boots
        case 22108:	callsub(S_Slot,22114);	break; // Modified_INT_Boots
        case 22109:	callsub(S_Slot,22115);	break; // Modified_AGI_Boots
        case 22110:	callsub(S_Slot,22116);	break; // Modified_VIT_Boots
        case 22111:	callsub(S_Slot,22117);	break; // Modified_DEX_Boots
        case 22112:	callsub(S_Slot,22118);	break; // Modified_LUK_Boots

        default:
            mes "[Artesăo de Hugin]";
            mes "Isso năo serve! Eu preciso de ^0000FFBotas Temporais^000000!";
            mes "O item precisa ser ^0000FFsem slot^000000!";
            close;
        }

    S_Slot:
        .@equip_id = getequipid(EQI_SHOES);
        mes "[Artesăo de Hugin]";
        mes "E saiba que ^FF0000todas as refinaçőes e encantamentos serăo perdidos.^000000";
        mes "Ainda quer arriscar?";
        next;
        if (select("Tudo bem, quero tentar!", "Cancelar") == 0) {
            mes "[Artesăo de Hugin]";
            mes "Tudo bem, volte quando quiser.";
            close;
        }
        if ((countitem(6607) < 20) || (countitem(6608) < 50)) {
            mes "[Artesăo de Hugin]";
            mes "Vocę precisa trazer ^0000FF20 Cristais Temporais e 50 Manas Coaguladas^000000. Nada é de graça, sabe...";
            close;
        }
        delitem 6607,20; // Temporal_Crystal

        if (getequipid(EQI_SHOES) == .@equip_id) {
            delequip EQI_SHOES;
            if (rand(1,100) < 20) {
                mes "[Artesăo de Hugin]";
                mes "Arrggh... falhamos. Melhor sorte da próxima vez.";
                specialeffect2 EF_PHARMACY_FAIL;
                close;
            }
            mes "[Artesăo de Hugin]";
            mes "Sucesso! Aqui está seu novo item.";
            specialeffect2 EF_MAXPOWER;
            getitem getarg(0),1;
        }
        close;
    end;
}

// Encantador de Botas Temporais
-	script	Mago de Hugin#pa0829	1_F_01,{

    OnTalk:
        disable_items;
        if (MaxWeight - Weight < 1000) {
            mes "Sua bolsa está muito pesada. Diminua o peso e volte.";
            close;
        }
        mes "[Artesăo de Hugin]";
        mes "Entăo vocę veio encantar as Botas Temporais, e o encantamento só é possível no par de botas que vocę está usando.";
        next;
        if (select("Encantar minhas Botas Temporais", "Como funciona o encantamento?") == 0) {
            mes "[Artesăo de Hugin]";
            mes "Vocę pode obter ^0000FFMana Coagulada^000000 durante suas jornadas no tempo.";
            next;
            mes "[Artesăo de Hugin]";
            mes "Eu posso encantar suas Botas Temporais dependendo da quantidade desse item. ^0000FFNăo é possível em botas comuns^000000, apenas nas 6 versőes especiais.";
            next;
            mes "[Artesăo de Hugin]";
            mes "Vocę pode escolher qual efeito deseja no item.";
            next;
            mes "[Artesăo de Hugin]";
            mes "Claro que isso vale apenas para o 4ş slot. Vocę começará com um efeito pequeno e ele ficará mais forte a cada aprimoramento.";
            next;
            mes "[Artesăo de Hugin]";
            mes "Para alcançar o efeito final, será necessário muito mais Mana Coaguladas. Năo se esqueça disso.";
            next;
            mes "[Artesăo de Hugin]";
            mes "Quando o 4ş slot atinge a opçăo máxima, vocę pode receber um efeito bônus aleatório no 3ş slot — mediante um pequeno pagamento~";
            next;
            mes "[Artesăo de Hugin]";
            mes "Ah, e năo se preocupe! Mesmo que o encantamento falhe, vocę năo perderá o item, nem os aprimoramentos ou cartas. Apenas năo podemos redefinir o encantamento.";
            next;
            mes "[Artesăo de Hugin]";
            mes "Ufa...";
            mes "Enfim, isso é tudo sobre o encantamento. Agora veja vocę mesmo como funciona.";
            close;
        }
        if (!getequipisequiped(EQI_SHOES)) {
            mes "[Artesăo de Hugin]";
            mes "Vocę está realmente usando o item?";
            close;
        }
        .@equip_id = getequipid(EQI_SHOES);

        setarray .@enchant_1[0],4808,4832,4814,4741,4869,4752; // Espírito de Luta 4 / Arqueiro Perito 1 / Feitiço 2 / Vitalidade 2 / Pós-Conjuraçăo 1 / Sorte 3
        setarray .@enchant_2[0],4820,4833,4813,4742,4872,4753; // Espírito de Luta 5 / Arqueiro Perito 2 / Feitiço 3 / Vitalidade 3 / Pós-Conjuraçăo 2 / Sorte 4
        setarray .@enchant_3[0],4821,4834,4812,4861,4873,4754; // Espírito de Luta 6 / Arqueiro Perito 3 / Feitiço 4 / HP Máx. 1 / Pós-Conjuraçăo 3 / Sorte 5
        setarray .@enchant_4[0],4822,4835,4826,4862,4881,4755; // Espírito de Luta 7 / Arqueiro Perito 4 / Feitiço 5 / HP Máx. 2 / Pós-Conjuraçăo 4 / Sorte 6
        setarray .@enchant_cost[0],1,4,15,30,50;

        switch(.@equip_id) {
        case 22000:	// Botas Temporais FOR
        case 22001:	// Botas Temporais INT
        case 22002:	// Botas Temporais AGI
        case 22003:	// Botas Temporais VIT
        case 22004:	// Botas Temporais DES
        case 22005:	// Botas Temporais SOR

        case 22006:	// Temporal_Str_Boots_ Slotada
        case 22007:	// Temporal_Vit_Boots_ Slotada
        case 22008:	// Temporal_Dex_Boots_ Slotada
        case 22009:	// Temporal_Int_Boots_ Slotada
        case 22010:	// Temporal_Agi_Boots_ Slotada
        case 22011:	// Temporal_Luk_Boots_ Slotada

            .@equip_name$ = getequipname(EQI_SHOES);
            setarray .@card[0],
                getequipcardid(EQI_SHOES,0),
                getequipcardid(EQI_SHOES,1),
                getequipcardid(EQI_SHOES,2),
                getequipcardid(EQI_SHOES,3);
            copyarray .@equip_card[0], .@card[0], 4; // Para checagem final
            .@equip_refine = getequiprefinerycnt(EQI_SHOES);

            if (.@card[2] > 0) {
                mes "[Artesăo de Hugin]";
                mes "Essas botas já atingiram o limite de encantamento. Năo posso encantá-las mais.";
                close;
            }
            if (.@card[3] == 0) { // Primeiro encantamento no 4ş slot
                .@cost = .@enchant_cost[0];
                mes "[Artesăo de Hugin]";
                mes "Quer encantar ^0000FF" + .@equip_name$ + "^000000?";
                mes "Para o primeiro encantamento, vocę precisa de ^0000FF" + .@cost + "^000000 Mana Coagulada.";
                next;
                .@s = select("Cancelar", "Espírito de Luta", "Perícia em Arco", "Feitiço", "Vitalidade", "Velocidade de Ataque", "Sorte") - 2;
                if (.@s == -1) {
                    mes "[Artesăo de Hugin]";
                    mes "Tudo bem, volte quando estiver pronto.";
                    close;
                }
                .@card[3] = .@enchant_1[.@s];
                .@string$ = "encantamento nível ^6300001^000000.";
            }
            else {
                for (.@enchant_num = 1; .@enchant_num < 5; .@enchant_num++) {
                    for (.@enchant_type = 0; .@enchant_type < 6 && .@card[3] != getd(".@enchant_" + .@enchant_num + "[" + .@enchant_type + "]"); .@enchant_type++)
                        continue;
                    if (.@enchant_type < 6)
                        break;
                }
                if (.@enchant_num == 5) {
                    mes "[Artesăo de Hugin]";
                    mes "Algo de errado aconteceu.";
                    close;
                }
                .@cost = .@enchant_cost[.@enchant_num];

                mes "[Artesăo de Hugin]";
                if (.@enchant_num == 4) {
                    .@card[2] = callfunc("F_Rand",4875,4876,4877,4878,4879,4880); // Poder do Urso, Magia Incontrolável, Velocidade da Luz, Tolo Musculoso, Olho de Águia, Dia de Sorte
                    .@string$ = "^990000bônus aleatório^000000 no slot 3.";
                    mes "O 4ş slot já foi aprimorado ao máximo. Deseja adicionar um efeito bônus aleatório ao 3ş slot? Vocę precisará de ^0000ff" + .@cost + "^000000 Mana Coagulada.";
                }
                else {
                    .@number = .@enchant_num + 1;
                    .@card[3] = getd(".@enchant_" + (.@enchant_num+1) + "[" + .@enchant_type + "]");
                    .@string$ = "encantamento nível ^630000" + .@number + "^000000.";
                    mes "Encantando o 4ş slot de ^0000FF" + .@equip_name$ + "^000000 para o nível ^0000FF" + .@number + "^000000. Requer ^0000FF" + .@cost + "^000000 Mana Coagulada.";
                }
                next;
                if (select("Aprimorar Encantamento!", "Cancelar") == 0) {
                    mes "[Artesăo de Hugin]";
                    mes "Tudo bem, volte quando estiver pronto.";
                    close;
                }
            }
            if (countitem(6608) < .@cost) {
                mes "[Artesăo de Hugin]";
                mes "Hmm... vocę está faltando " + (.@cost - countitem(6608)) + " Mana Coagulada. Traga mais e poderemos continuar.";
                close;
            }
            specialeffect2 EF_REPAIRWEAPON;
            delitem 6608,.@cost; // Feitiço Coagulado

            // Anti-hack
            if (callfunc("F_IsEquipIDHack", EQI_SHOES, .@equip_id) || callfunc("F_IsEquipCardHack", EQI_SHOES, .@equip_card[0], .@equip_card[1], .@equip_card[2], .@equip_card[3]) || callfunc("F_IsEquipRefineHack", EQI_SHOES, .@equip_refine))
                close;

            delequip EQI_SHOES;
            mes "[Artesăo de Hugin]";
            mes "Tentando " + .@string$;
            getitem2 .@equip_id,1,1,.@equip_refine,0,0,0,.@card[2],.@card[3];
            close;
        default:
            mes "[Artesăo de Hugin]";
            mes "Este năo é o item correto para encantamento.";
            mes "Lembre-se, apenas as 6 Botas Temporais podem ser encantadas.";
            close;
        }
    end;
}
